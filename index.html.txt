<script>
function openCard() {
    hideAll();
    document.getElementById('birthdayCard').classList.remove('hidden');
}

function showMemory() {
    hideAll();
    document.getElementById('memory').classList.remove('hidden');
}

function showWhy() {
    hideAll();
    document.getElementById('why').classList.remove('hidden');
}

function showSecret() {
    hideAll();
    document.getElementById('secret').classList.remove('hidden');
}

function back() {
    hideAll();
    document.getElementById('birthdayCard').classList.remove('hidden');
}

function hideAll() {
    document.querySelectorAll('.card').forEach(card => card.classList.add('hidden'));
}

function celebrate() {
    for (let i = 0; i < 50; i++) {
        let confetti = document.createElement('div');
        confetti.classList.add('confetti');
        confetti.style.left = Math.random() * 100 + "vw";
        confetti.style.backgroundColor = ["#ffb6c1","#ffffff","#ff69b4"][Math.floor(Math.random()*3)];
        confetti.style.animationDuration = (Math.random() * 2 + 2) + "s";
        document.body.appendChild(confetti);

        setTimeout(() => {
            confetti.remove();
        }, 3000);
    }
}
